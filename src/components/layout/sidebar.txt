import React from "react";
import {
  IconButton,
  Typography,
  List,
  ListItem,
  ListItemPrefix,
  Drawer,
  Card,
} from "@material-tailwind/react";
import {
  PresentationChartBarIcon,
  ShoppingBagIcon,
  UserCircleIcon,
  Cog6ToothIcon,
  InboxIcon,
  PowerIcon,
} from "@heroicons/react/24/solid";
import { Bars3Icon } from "@heroicons/react/24/outline";

function Sidebaradmin() {
  const [isDrawerOpen, setIsDrawerOpen] = React.useState(false);
  const [showBarsIcon, setShowBarsIcon] = React.useState(true);

  const openDrawer = () => {
    setShowBarsIcon(false); // Ocultar el icono de las barras
    setIsDrawerOpen(true);
  };

  const closeDrawer = () => {
    setIsDrawerOpen(false);
    setTimeout(() => setShowBarsIcon(true), 500);
  }


  return (
    <>
      {showBarsIcon && (
        <IconButton variant="text" size="lg" onClick={openDrawer}>
          <Bars3Icon className="h-8 w-8 stroke-2" />
        </IconButton>
      )}
      <Drawer open={isDrawerOpen} onClose={closeDrawer}>
        <Card
          color="transparent"
          shadow={true}
          className="h-[calc(100vh-2rem)] w-full p-4"
        >
          <div className="mb-2 flex items-center gap-4 p-4">
            <img
              src="https://docs.material-tailwind.com/img/logo-ct-dark.png"
              alt="brand"
              className="h-8 w-8"
            />
            <Typography variant="h5" color="blue-gray">
              TaskBoard
            </Typography>
          </div>
          <List>
            <ListItem>
              <ListItemPrefix>
                <PresentationChartBarIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2 font-semibold">Home</p>
            </ListItem>
            <ListItem>
              <ListItemPrefix>
                <ShoppingBagIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2">Usuarios</p>
            </ListItem>
            <ListItem>
              <ListItemPrefix>
                <InboxIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2 font-semibold">Roles</p>
            </ListItem>
            <ListItem>
              <ListItemPrefix>
                <UserCircleIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2 font-semibold">Profile</p>
            </ListItem>
            <ListItem>
              <ListItemPrefix>
                <Cog6ToothIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2 font-semibold">Settings</p>
            </ListItem>
            <ListItem>
              <ListItemPrefix>
                <PowerIcon className="h-5 w-5" />
              </ListItemPrefix>
              <p className="ml-2 font-semibold ">Log Out</p>
            </ListItem>
          </List>
        </Card>
      </Drawer>
    </>
  );
}

export default Sidebaradmin
